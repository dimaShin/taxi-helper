 12 define(['app', 'async!googleMapsApi'], function(app){
 13     var geocoder = new google.maps.Geocoder();
 14     app.factory('addressService', function($interval){
 15 
 16  /**
 17  *@function getLatLng
 18  *@param {object} points
 19  * converts points into google maps LatLng object
 20  *@return promise
 21  */
 22         function getLatLng(points){
 23             var deferred = $.Deferred(),
 24                 pointLatLng = {
 25                     from: null,
 26                     to: null,
 27                     waypoints: []
 28                 };
 29             for(var i in points){
 30                 (function(i){
 31                     if(i === 'waypoints'){
 32                         for(var j = 0; j < points.waypoints.length; j++){
 33                             geocoder.geocode( { 'address': 'kharkov, ukraine, ' + points.waypoints[j].value}, function(results, status) {
 34                                 if (status == google.maps.GeocoderStatus.OK) {
 35                                     pointLatLng.waypoints.push({
 36                                         location: results[0].geometry.location,
 37                                         stopover: true
 38                                     });
 39                                 } else {
 40                                     deferred.reject(status);
 41                                     $interval.cancel(interval);
 42                                 }
 43                             });
 44                           }
 45                     }else{
 46                         if(points[i][0] === '('){
 47                             var latLng = points[i].replace(/[\(\)\s]/g, '').split(','),
 48                                 lat = latLng[0],
 49                                 lng = latLng[1];
 50 
 51                             pointLatLng[i] = new google.maps.LatLng(lat, lng);
 52                             console.log('lat/lng: ', pointLatLng[i]);
 53                         }else{
 54                             geocoder.geocode( { 'address': 'kharkov, ukraine, ' + points[i]}, function(results, status) {
 55                                 if (status == google.maps.GeocoderStatus.OK && results[0].address_components.length > 5){
 56                                     pointLatLng[i] = results[0].geometry.location;
 57                                 } else {
 58                                     deferred.reject(status);
 59                                     $interval.cancel(interval);
 60                                 }
 61                             });
 62                         }
 63 
 64                     }
 65                 })(i)
 66             }
 67             var interval = $interval(function(){
 68                 if(pointLatLng.from && pointLatLng.to && pointLatLng.waypoints.length === points.waypoints.length){
 69                     deferred.resolve(pointLatLng);
 70                     $interval.cancel(interval);
 71                 }
 72             }, 100)
 73             return deferred.promise();
 74         }
